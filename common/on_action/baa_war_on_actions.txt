# Better Automated Armies — army hooks (Stage 1).
#
# on_army_monthly (vanilla army_on_actions.txt, header before the block):
#   root = owner of the army
#   scope:army = the army
#
# Role assignment runs on the army owner when they are at war as a primary attacker.
# Idempotency lives in baa_assign_all_roles_effect (baa_role_assigned + commander lists).
#
# Stage 2: add strategy tick logic inside baa_on_army_monthly_role_assignment BELOW
# baa_assign_all_roles_effect = yes. Same pulse order is safe: assignment runs first and sets
# baa_role_assigned; tick code should gate on that (and Stage 2 vars). Army-monthly is ~30 days
# and staggered by army ID, so a stack that just gained commanders typically gets roles on this
# pulse and tick-relevant behavior on a later pulse once flags/state are stable.

on_army_monthly = {
	on_actions = {
		baa_on_army_monthly_role_assignment
	}
}

baa_on_army_monthly_role_assignment = {
	effect = {
		root = {
			if = {
				limit = {
					is_at_war = yes
					any_character_war = {
						primary_attacker = root
					}
				}
				baa_assign_all_roles_effect = yes
			}
		}
	}
}
